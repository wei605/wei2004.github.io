#需下載套件
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from webdriver_manager.chrome import ChromeDriverManager
import time
import pandas as pd

Step1 抓取菜單中麵類品項 


#抓取菜單中麵類品項
def get_menu():
    options = webdriver.ChromeOptions()
    options.add_argument("--disable-blink-features=AutomationControlled")

    driver = webdriver.Chrome(
        service=webdriver.chrome.service.Service(ChromeDriverManager().install()),
        options=options
    )

    try:
        driver.get("https://www.mkkramen.com/food")

Step2 抓取菜單中的li.menu-item 


	# 等待 li.menu-item 出現
        WebDriverWait(driver, 15).until(
            EC.presence_of_element_located((By.CSS_SELECTOR, "li.menu-item"))
        )
	# 緩衝時間確保數據完全渲染
        time.sleep(2) 
        
	# 抓取所有菜單項目每個 li.menu-item
        cards = driver.find_elements(By.CSS_SELECTOR, "li.menu-item")
	# 存放菜單資訊
        menu = []

Step3 先抓整段 textContent，是因為品項名稱和價格混在一起，不是分開的欄位。透過抓全部文字再扣掉價格，才能正確取得品項名稱。 


for card in cards:
            try:
		# 抓價格
                price = card.find_element(By.CSS_SELECTOR, ".price").text.strip()
                
                # 用 textContent 抓全部文字，然後去掉價格得到名稱
                full_text = card.get_attribute("textContent").strip()
                title = full_text.replace(price, "").strip()
                
                # 只保留名稱中包含「麵」的項目
                if "麵" in title:
                   menu.append({"品項": title, "價格": price})
            except Exception as e:
                print(f"跳過一項錯誤: {e}") 
        return menu

    except Exception as e:
        print(f"發生錯誤: {e}")
        driver.save_screenshot("error_menu.png")
        return None
    finally:
        driver.quit()

Step4 輸出只有麵類的品項及價格 


if __name__ == "__main__":
    result = get_menu()
    if result:
        print(f"成功獲取 {len(result)} 品項：\n")
        for item in result:
            print(f"{item['品項']} - ${item['價格']}")
    else:
        print("查無資料")

爬蟲成果

這次的專案成果是利用 Python 結合 Selenium 技術，成功爬取 麵框框官方網站上的菜單資訊。透過模擬使用者操作，我的程式能夠進入網站並等待網頁的內容完整載入後，擷取出所有的菜單項目。為了聚焦於主要品項，我進一步篩選所有菜單中包含「麵」這個字的品項，也就是只擷取拉麵類的餐點，並搭配價格資料一併整理呈現。
